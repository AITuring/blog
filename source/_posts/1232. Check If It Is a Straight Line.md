class Solution:
    def checkStraightLine(self, coordinates: List[List[int]]) -> bool:
        k = []
        for i in range(1,len(coordinates)):
            count=1
            dy= (coordinates[i][1]-coordinates[i-1][1])
            dx=(coordinates[i][0]-coordinates[i-1][0])
            if dx==0:
                count=0
            else:count=dy/dx
            k.append(count)
        k=set(k)
        if len(k)==1:
            return True
        else:return False
